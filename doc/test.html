  35.     private String url;
  36.     @Value("${ds1.datasource.username}")
  37.     private String username;
  38.     @Value("${ds1.datasource.password}")
  39.     private String password;
  40.     @Value("${ds1.datasource.driverClassName}")
  41.     private String driverClassName;
  42.  
  43.     @Value("${ds1.datasource.maxActive}")
  44.     private Integer maxActive;
  45.     @Value("${ds1.datasource.minIdle}")
  46.     private Integer minIdle;
  47.     @Value("${ds1.datasource.initialSize}")
  48.     private Integer initialSize;
  49.     @Value("${ds1.datasource.maxWait}")
  50.     private Long maxWait;
  51.     @Value("${ds1.datasource.timeBetweenEvictionRunsMillis}")
  52.     private Long timeBetweenEvictionRunsMillis;
  53.     @Value("${ds1.datasource.minEvictableIdleTimeMillis}")
  54.     private Long minEvictableIdleTimeMillis;
  55.     @Value("${ds1.datasource.testWhileIdle}")
  56.     private Boolean testWhileIdle;
  57.     @Value("${ds1.datasource.testWhileIdle}")
  58.     private Boolean testOnBorrow;
  59.     @Value("${ds1.datasource.testOnBorrow}")
  60.     private Boolean testOnReturn;
  61.  
  62.     // spring-boot不是原生支持的druid连接池,
  63.     // spring-boot不支持自动配druid连接池（即无法通过配置项直接支持相应的连接池连接池）
  64.     @Bean(name = "dataSource") //DataSource对象被@Bean声明，为Spring容器所管理
  65.     @Primary //@Primary表示这里定义的DataSource将覆盖其他来源的DataSource
  66.     public DataSource dataSource() {
  67.         //jdbc配置
  68.         DruidDataSource dataSource = new DruidDataSource();
  69.         dataSource.setDriverClassName(driverClassName);
  70.         dataSource.setUrl(url);
  71.         dataSource.setUsername(username);
  72.         dataSource.setPassword(password);
  73.  
  74.         //连接池配置
  75.         dataSource.setMaxActive(maxActive);
  76.         dataSource.setMinIdle(minIdle);
  77.         dataSource.setInitialSize(initialSize);
  78.         //配置获取连接等待超时的时间
  79.         dataSource.setMaxWait(maxWait);
  80.         //配置间隔多久才进行一次检测，检测需要关闭的空闲连接，单位是毫秒
  81.         dataSource.setTimeBetweenEvictionRunsMillis(timeBetweenEvictionRunsMillis);
  82.         //配置一个连接在池中最小生存的时间，单位是毫秒
  83.         dataSource.setMinEvictableIdleTimeMillis(minEvictableIdleTimeMillis);
  84.         //申请连接的时候检测，如果空闲时间大于timeBetweenEvictionRunsMillis，检测连接是否有效
  85.         dataSource.setTestWhileIdle(testWhileIdle);
  86.         //申请连接时,检测连接是否有效
  87.         dataSource.setTestOnBorrow(testOnBorrow);
  88.         //归还连接时,检测连接是否有效
  89.         dataSource.setTestOnReturn(testOnReturn);
  90.         //测试连接是否有效的sql
  91.         dataSource.setValidationQuery("select 1 from dual");
  92.  
  93.         //打开PSCache，并且指定每个连接上PSCache的大小
  94.         dataSource.setPoolPreparedStatements(true);
  95.         dataSource.setMaxPoolPreparedStatementPerConnectionSize(20);
  96.  
  97.         try {
  98.              /*
  99.             配置监控统计拦截的filters，去掉后监控界面sql无法统计，'wall'用于防火墙
  100.             # 属性类型是字符串，通过别名的方式配置扩展插件，常用的插件有：
  101.             # 监控统计用的filter:stat
  102.             # 日志用的filter:log4j
  103.             # 防御sql注入的filter:wall
  104.             spring.datasource.filters=stat,log4j,wall
  105.             */
  106.             dataSource.setFilters("stat,wall");
  107.         } catch (SQLException e) {
  108.             e.printStackTrace();
  109.         }
  110.         return dataSource;
  111.     }

  126.         sessionFactory.setMapperLocations(new PathMatchingResourcePatternResolver()
  127.                 .getResources(DatasourceConfig.MAPPER_LOCATION));
  128.         return sessionFactory.getObject();
  129.     }

  135.         //设置控制台管理url http://localhost:8080/druid/
  136.         servletRegistrationBean.addUrlMappings("/druid/*");
  137.         Map<String, String> initParameters = new HashMap<String, String>();
  138.         //设置控制台管理用户
  139.         initParameters.put("loginUsername", "admin");// 用户名
  140.         initParameters.put("loginPassword", "admin");// 密码
  141.         initParameters.put("resetEnable", "false");// 禁用HTML页面上的“Reset All”功能
  142.         initParameters.put("allow", ""); // IP白名单 (没有配置或者为空，则允许所有访问)
  143.         //initParameters.put("deny", "192.168.20.38");// IP黑名单 (存在共同时，deny优先于allow)
  144.         servletRegistrationBean.setInitParameters(initParameters);
  145.         return servletRegistrationBean;
  146.     }

  149.     public FilterRegistrationBean filterRegistrationBean() {
  150.         //创建过滤器

  157.         return filterRegistrationBean;
